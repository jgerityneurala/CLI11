550 inline int64_t to_flag_value(std::string val) {
551  static const std::string trueString("true");
552  static const std::string falseString("false");
553  if(val == trueString) {
554  return 1;
555  }
556  if(val == falseString) {
557  return -1;
558  }
559  val = detail::to_lower(val);
560  int64_t ret;
561  if(val.size() == 1) {
562  if(val[0] >= '1' && val[0] <= '9') {
563  return (static_cast<int64_t>(val[0]) - '0');
564  }
565  switch(val[0]) {
566  case '0':
567  case 'f':
568  case 'n':
569  case '-':
570  ret = -1;
571  break;
572  case 't':
573  case 'y':
574  case '+':
575  ret = 1;
576  break;
577  default:
578  throw std::invalid_argument("unrecognized character");
579  }
580  return ret;
581  }
582  if(val == trueString || val == "on" || val == "yes" || val == "enable") {
583  ret = 1;
584  } else if(val == falseString || val == "off" || val == "no" || val == "disable") {
585  ret = -1;
586  } else {
587  ret = std::stoll(val);
588  }
589  return ret;
590 }
